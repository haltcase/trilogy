new Trilogy(fileName, optsopt)

Source:
  • index.js, line 52

Initialize a new datastore instance, creating an SQLite database file at the fileName path if it does not yet exist, or reading it if it does.

Example
import Trilogy from 'trilogy'

const db = new Trilogy('./storage.db')

// WITH OPTIONS:
// verbose function
const db = new Trilogy('./storage.db', {
  verbose: console.log.bind(console)
})

// errorListener function
function errorHandler (err) {
  if (err.message === `Trilogy#createTable :: 'columns' must be an array`) {
    console.log('Crap. Should have read the docs!')
  }
}

const db = new Trilogy('./storage.db', {
  errorListener: errorHandler
})

createTable(tableName, columns, optionsopt) → {Promise}

Source:
  • index.js, line 204

Add a table to the database. The columns argument must be an array of either strings or Objects, which can be mixed and matched. String values default to a column of the SQLite 'text' type. If another type is needed or any other attributes, use an Object.

All the properties of the supplied column Object are passed to knex. Some attributes require no values, such as primary or nullable. In these cases, their presence in the object is enough to add that flag.

If the column property is not present in knex's methods it will be ignored. See knex's documentation on Schema Building for the available attributes when creating column tables.

Example
// `columns` should be an Array
// each item in the Array should be either a string or an Object

// a string in the Array defaults to a text column in the table
db.createTable('people', ['email'])

// use an object to specify other attributes
db.createTable('people', [
  { name: 'age', type: 'integer' }
])

// you can mix and match
db.createTable('people', [
  'name',
  { name: 'age', type: 'integer' },
  'email',
  // note that the value of `primary` doesn't make a difference
  // this would still be a primary key column
  { name: '', primary: false }
])

where Object | Array | function <optional>

Defaults to no restriction on selection.
allowNegative boolean

Unless set to true, the value will not be allowed to go below a value of 0.

first(table, columnsopt, whereopt, optionsopt) → {Promise.<Object>}

Source:
  • index.js, line 410

Return the first row selected by the query. Allows overloading of arguments, ie. table is the only required argument. In this case, columns defaults to selecting all columns.

Example
// select the first record in 'people'
db.first('people')

// select a random record from 'people'
// the second argument is a where clause but is always true for all records
db.first('people', ['1', '=', '1'], { random: true })

// NOTE:
// even with overloading, in this case `where` needs to be provided if we
// have an `options` object. this is because `where` could also be an object
// so the function has no way to know which one you meant to provide.

db.getValue('people.age', { name: 'Bob' })
db.getValue('people[age]', { name: 'Bob' })

select(table, columnsopt, whereopt, optionsopt) → {Promise.<Array>}

Source:
  • index.js, line 337

Execute a select query on the database. Allows overloading of arguments, ie. table is the only required argument. In this case, columns defaults to selecting all columns.

Example
// select all records in the 'people' table
db.select('people')

// select just the 'age' and 'favColor' columns where name is 'Bob'
db.select('people', ['age', 'favColor'], { name: 'Bob' })

// select just 'name' where age is at least 18
db.select('people', 'age', ['age', '>=', '18'])
